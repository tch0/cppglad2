cmake_minimum_required(VERSION 3.10)

project (glad
    DESCRIPTION "glad generated by https://gen.glad.sh/"
    VERSION 2.0.3
)

add_library(glad STATIC src/gl.c)

target_include_directories(glad
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}/include
)

install(TARGETS glad DESTINATION lib)
install(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/include/glad DESTINATION include)
install(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/include/KHR DESTINATION include)
